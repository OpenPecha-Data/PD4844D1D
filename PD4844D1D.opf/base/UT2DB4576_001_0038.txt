སྨད་འདུལ་གྱི་བཤད་རྒྱུད་བཞུགས་སོ།། 

 ན་མོ་གུ་རུ་བྷྱ། སྨད་འདུལ་དུ་གྲགས་པའི་བཤད་རྒྱུད་ནི། སྟོན་པ་སངས་རྒྱས་ཏེ། ཉེ་བར་འཁོར། ཀུན་དགའ་བོ་དང༌། ཤ་ནའི་གོས་ཅན། ཉེར་སྦས་དྷི་དྷི་ཀ ནག་པོ་ལེགས་མཐོང༌། བློ་གྲོས་མི་འབེབས། ཡོན་ཏན་འོད། ཤཱ ཀྱ་འོད། ཆོས་ཀྱི་བཤེས་གཉེན། ཆེ་ཆུང་གཉིས། དེ་གཉིས་ཀས་སེང་གེ་གདོང་ཅན། དེས་དྷ་ན་ཤཱི་ལ་དང༌། ཛཱི་ན་མི་ཏྲ། དེས་ལྕོག་རོ་ཀླུའི་རྒྱལ་མཚན། དེས་གཡོ་དགེ་འབྱུང་དང༌། གཙང་རབ་གསལ། དམར་ ཤཱ ཀྱ་མུ་ནེ་གསུམ་ལ་བཤད། དེ་གསུམ་མདོ་སྨད་དུ་བྱོན་ནས། ཨ་མེས་བླ་ཆེན་དགོང་པ་རབ་གསལ་ལ་བཤད། དེས་སློབ་མ་ལ་སྔ་བ་ཁམས་ཀྱི་ཟོག་པོ་མི་དྲུག་བྱ་བ་བྱུང་ཏེ། སྤང་གོང་ཡེ་ཤེས་གཡུང་དྲུང༌། པར་རིན་ཆེན་གསལ། འལ་རྡོ་རྗེ་དབང་ཕྱུག སྣུབས་རཀྴི་ཏ། ཇོ་བོ་ཡེ་ཤེས་སེང་གེ ལྕོག་རོ་ཤེས་རབ་བྱང་ཆུབ་རྣམས་ཡིན་པ། དང་པོ་མཁན་སློབ་ཆ་མ་ཚང་བས། བསྙེན་རྫོགས་མ་གྲུབ། དགེ་ཚུལ་ཡིན། འདུལ་བ་ལ་སོགས་ཆོས་རྣམས་ནི་གནང༌། བླ་ཆེན་གྱི་བོན་པོའི་ཐུགས་ཞེན་དུ་བྱས་ནས། ཞྭ་ཐེབ་ཀྱི་མུ་ཐུགས་སྔོན་པོར་བྱེད་པ་སོགས་ཀྱི། བཀའ་རྟགས་ཀྱི་ཕྱག་རྒྱ་བཞི་བྱ་བ་ཡང་ཁོང་དྲུག་པོ་ལ་གནང་བར་བྱེད་དོ། །སྤང་པར་གཉིས་ཀྱི་སློབ་མ། གྲུམ་ཡེ་ཤེས་རྒྱལ་མཚན། དེས་གྲུམ་གཡུང་དྲུང་རྒྱལ་མཚན་ཤིང་ལག་ཅན་ལ། དེས་དབུས་གཙང་གི་མི་བཅུའི་ནང་ནས། ཀླུ་མེས་ཚུལ་ཁྲིམས་ཤེས་རབ་ལ་འདུལ་བ་བཤད། དེས་གཟུས་རྡོ་རྗེ་རྒྱལ་མཚན་ལ། དེ་ནས་རིམ་པ་བཞིན་དུ། སྣེལ་བོ་གྲགས་པ་རྒྱལ་མཚན། རྒྱ་དབང་རྡོར། བྱ་བརྩོན་འགྲུས་འབར། རྩིས་པ་གཞོན་སེང༌། སྙེར་ཆུང་བཟང་སྟོན། དེ་ལ་ ཤཱ ཀྱའི་དགེ་སློང་ཐུབ་པ་ཤེས་རབ་བདག་གི་འདུལ་བའི་ཆོས་སྐོར་རྣམས་ཚང་མར་ཞུས་པ་ལེགས་སོ། །ཡང་གྲུམ་ཤིང་ལག་ཅན་གྱིས། ལོ་སྟོན་རྡོ་རྗེ་དབང་ཕྱུག་དང༌། ཚོང་བཙུན་ཤེས་རབ་སེང་གེ སུམ་པ་ཡེ་ཤེས་བློ་གྲོས། ཨ་བཙུན་ ཤཱ ཀྱ་རྒྱལ་མཚན། འབྲེ་སྟོན་ཚུལ་ཁྲིམས་གཞོན་ནུ་རྣམས་ལ། འདུལ་བ་བཤད་ནས་དབུས་གཙང་དུ་དར་བར་མཛད་པ་རྣམས་ལའང༌།
