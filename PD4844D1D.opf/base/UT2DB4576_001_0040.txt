ཚོགས་པ་སྡཻ་བཞིར་གྱེས་ཚུལ་བཞུགས་སོ། །

ན་མོ་སངྒ་ཡ། ཐུབ་བསྟན་རྩོད་ལྡན་དུས་འདིར་གསལ་མཛད་པའི།། པཎ་ཆེན་མཁན་རྒྱུད་ཚོགས་པ་སྡཻ་བཞིའི་ཁུངས། །སྟོན་པ་སངས་རྒྱས་དངོས་སློབ་ཤཱ་རིའི་བུ། །སྒྲ་གཅན་འཛིན་གཉིས་ཀླུ་སྒྲུབ་གྷུ་མ་ཏི། །རཏྣ་མི་ཏྲ་དྷརྨ་ཕ་ལ་དང༌།། གྷུ་ན་པ་ཏི་ཆོས་འཕྲེང་འབྱུང་གནས་སྦས། །དེས་སློབ་པཎ་ཆེན་ཤཱཀྱ་ཤྲཱི་ཞེས་པ།། སྟོན་པ་སྐུ་འདས་སུམ་སྟོང་ཉི་བརྒྱ་དང༌། །ལྔ་བཅུ་རྩ་བརྒྱད་ཆུ་ཕོ་སྟག་ལ་བལྟམས། །བརྒྱད་ཅུ་ལྔ་བཞེས་མེ་སྟག་དབུས་སུ་ཕེབས། །ལོ་གྲངས་བཅུ་གཉིས་བོད་དུ་འགྲོ་དོན་མཛད། །དགུ་བཅུ་བདུན་ལ་རྒྱ་གར་ཡུལ་ལ་བྱོན། །གོ་དགུ་ལྕགས་ཕོ་འབྲུག་ལ་མྱ་ངན་འདས། །པཎ་ཆེན་ཐེགས་ནས་ལོ་དགུར་ཙམ་སོང་བས། །གདན་གཅིག་སྲོལ་བསྐྱངས་དེ་རྟིང་རྡོ་རྗེ་དཔལ། །སངས་རྒྱས་དཔལ་ནས་གཞུང་པ་དཔོན་རྒྱལ་གྱེས། །བྱེ་རྫིང་པར་གྲགས་དེ་ནས་ཐུགས་རྗེ་དཔལ།། དར་མ་དཔལ་བཟང་གཞོན་ནུ་བཟང་ཚུལ་དཔལ། །སངས་གཞོན་དར་བཟང་ཚུལ་ཁྲིམས་མགོན་པོ་པ། །བསོད་འོད་རིན་ཚུལ་བསོད་ཤེས་ཤེས་རབ་མགོན།། གྲགས་བསོད་ཤེར་འཕགས་ཤེར་ཚུལ་རིན་ཤེས་པ། །ཆོས་དཔལ་ཤེར་འོད་ཀུན་དགའ་ཤེས་རབ་ཡུན། །རྡོར་དཔལ་མན་ལ་མཁན་རབ་ཉེར་གཉིས་བྱོན། །པཎ་ཆེན་མེ་སྟག་ཕེབས་ནས་དིང་སང་གིས། །ཤིང་སྟག་བརྒྱད་དཀར་དབུས་གནས་ཡར་ཆོད་ལ། །ལོ་གྲངས་ཉིས་བརྒྱ་བཞི་བཅུ་རྩ་བདུན་སོང༌། །ཚལ་མིག་པ་ལ་མཁན་པོ་སངས་རྒྱས་དཔལ། །འོད་དཔལ་ཆོས་ཀྱི་རྒྱལ་མཚན་སངས་རིན་པ།། འོད་དཔལ་ཤེར་མགོན་སེང་རྒྱལ་བསོད་དབང་དང༌། །དྲན་ཚུལ་ཚུལ་རིན་དཔང་བློ་རྒྱལ་དབང་དང༌། །སྲས་སེང་ཡོན་རིན་བྱང་ཆུབ་བཟང་པོ་ཡན། །བཅུ་ཕྲག་ཕྱེད་དང་གཉིས་པོ་དིང་སང་དུས། །ཚོགས་ཆེན་པར་གྲགས་ཐུབ་བསྟན་དྲི་མེད་ཀྱིས། །སྲོག་འཛིན་མཁན་རྒྱུད་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ལུང་ཐོག་མ་ལྷོ་བྲག་བྱང་ཆུབ་དཔལ། །བསོད་སྟོབས་སྡེ་དཔལ་གྲགས་གཞོན་བསོད་ནམས་གྲགས། །བསོད་བཟང་གཞོན་མགོན་མཁན་ཆེན་གྲགས་རྒྱལ་དང༌། །གྲགས་ཚུལ་གྲགས་པ་བཤེས་གཉེན་རྣམ་རྒྱལ་བ།

།འཇམ་དབྱངས་རིན་རྒྱལ་རྒྱལ་མཆོག་གྲགས་དབང་རྒྱལ། །ཟླ་བློ་རྒྱལ་བ་ཕྱག་ན་དིང་སང་ཡན། །བཅུ་ཕྲག་གཅིག་དང་དྲུག་ཅུ་བྱུང་བར་གྲགས། །སླབ་གསུམ་རྣམ་དག་ཤཱཀྱའི་བསྟན་པ་ཡི།། གསལ་བྱེད་མཁན་རྒྱུད་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །དགེ་འདུན་སྒང་པ་བྱང་ཆུབ་དཔལ་ནས་གྱེས། །དབང་གྲགས་གཞོན་རྒྱལ་འདུལ་ཚ་དོན་གྲུབ་དཔལ། །ཡོན་ཏན་རྒྱ་མཚོ་དཔལ་གྲུབ་ཡོན་བློ་བ། བརྩོན་རྒྱལ་སེང་གེ་དཔལ་དང་ཆོས་གྲུབ་ཡན། །བཅུ་དང་གཅིག་གིས་འགྲོ་ཀུན་སྨིན་གྲོལ་ལ། །འགོད་མཛད་མཁན་རྒྱུད་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །སྙིགས་མའི་དུས་འདིར་འགྲོ་མང་པདྨོའི་གཉེན།། བོད་ཡུལ་མུན་པ་སེལ་བར་མཛད་པ་ཡིས། །པཎ་ཆེན་མཁན་རྒྱུད་རིམ་པའི་འོད་ཟེར་གྱིས། །སངས་རྒྱས་བསྟན་པ་དར་མཛད་འཕྲིན་ལས་པ། ཁ་ཆེ་པཎ་ཆེན་སློབ་རྒྱུད་འདི་བྲིས་པས། ཁ་བ་ཅན་འདིར་ཐུབ་བསྟན་སེང་གེའི་སྒྲས། ཁ་ན་མ་ཐོ་ཆོས་ཀུན་ཟིལ་མནན་ནས། ཁ་ཊྭཾ་གྱིས་མཚོན་སྐུ་གསུམ་མྱུར་ཐོབ་ཤོག། །། དགེ་ལེགས་འཕེལ།། །། མེས་པོའི་ཤུལ་བཞག་ལས། རྒྱལ་རབས་མང་པོའི་ལེགས་བཤད་རྣམ་གྲངས་ཡིད་འཛིན་ནོར་བུའི་ཕྲེང་བ།ཚོགས་པ་སྡཻ་བཞིར་གྱེས་ཚུལ། -PAGE 252--PAGE 271--PAGE 251-
